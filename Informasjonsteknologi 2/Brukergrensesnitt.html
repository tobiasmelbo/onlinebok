import streamlit as st
from math import sqrt

st.title("Kalkulator")

            svar = 1/x
    
    with cols[1]:

            svar = x**2

    b = st.number_input("Velg et tall b", step=1)
    svar = None

    cols = st.columns(5)

    with cols[0]:
        if st.button("a + b"):
            svar = a + b
    
    with cols[1]:
        if st.button("a - b"):
            svar = a - b
    
    with cols[2]:
        if st.button("a * b"):
